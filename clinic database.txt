
CREATE TABLE StatusInfo(
StatusID INT AUTO_INCREMENT NOT NULL,
StatusName VARCHAR(20) NOT NULL,
PRIMARY KEY (StatusID)
);

CREATE TABLE PersonInfo(
PID INT NOT NULL,
FirstName VARCHAR(20) NOT NULL,
LastName VARCHAR(20)  NOT NULL,
OtherNames VARCHAR(30),
DateOfBirth DATE      NOT NULL,
Height     DECIMAL,
Weight     DECIMAL,
PriorIssues VARCHAR(50) NOT NULL,
KnownAllergies VARCHAR(50),
StatusID  INT,
PRIMARY KEY (PID),
FOREIGN KEY (StatusID) REFERENCES StatusInfo (StatusID)
);


CREATE TABLE UsersInfo (
UID INT NOT NULL,
FirstName VARCHAR(20) NOT NULL,
LastName VARCHAR(20) NOT NULL,
Username VARCHAR(20) NOT NULL,
PRIMARY KEY (UID)
);

CREATE TABLE VisitLogs(
VID INT AUTO_INCREMENT NOT NULL,
PID INT NOT NULL,
UID INT NOT NULL,
DateOfVisit DATE NOT NULL,
Observations VARCHAR(50) NOT NULL,
VitalsInfo VARCHAR(50),
Symptons VARCHAR(50),
PRIMARY KEY (VID),
FOREIGN KEY (PID) REFERENCES PersonInfo (PID)

);
